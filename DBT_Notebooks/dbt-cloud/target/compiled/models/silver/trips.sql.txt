




SELECT
    
        trip_id
         
             ,
         
    
        vehicle_id
         
             ,
         
    
        customer_id
         
             ,
         
    
        driver_id
         
             ,
         
    
        trip_start_time
         
             ,
         
    
        trip_end_time
         
             ,
         
    
        start_location
         
             ,
         
    
        end_location
         
             ,
         
    
        distance_km
         
             ,
         
    
        fare_amount
         
             ,
         
    
        payment_method
         
             ,
         
    
        trip_status
         
             ,
         
    
        last_updated_timestamp
         
    
FROM 
    `pysparkdbt`.`bronze`.`trips`

WHERE 
    last_updated_timestamp > (SELECT coalesce(MAX(last_updated_timestamp),'1900-01-01') FROM `pysparkdbt`.`silver`.`trips`)
